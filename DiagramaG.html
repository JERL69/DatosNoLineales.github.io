Diagramas de Flujo - Grafos

1. Recorrido en anchura (BFS)

        graph TD;
            A[Inicio] --> B[Crear una cola vacía];
            B --> C[Marcar nodo de inicio como visitado];
            C --> D[Encolar nodo de inicio];
            D --> E[¿Cola vacía?];
            E -->|Sí| F[Fin];
            E -->|No| G[Desencolar nodo];
            G --> H[Explorar vecinos];
            H --> I{¿Vecino visitado?};
            I -->|No| J[Marcar como visitado y encolar];
            J --> H;
            I -->|Sí| E;

2. Recorrido en profundidad (DFS)

        graph TD;
            A[Inicio] --> B[Crear una pila vacía];
            B --> C[Marcar nodo de inicio como visitado];
            C --> D[Apilar nodo de inicio];
            D --> E[¿Pila vacía?];
            E -->|Sí| F[Fin];
            E -->|No| G[Desapilar nodo];
            G --> H[Explorar vecinos];
            H --> I{¿Vecino visitado?};
            I -->|No| J[Marcar como visitado y apilar];
            J --> H;
            I -->|Sí| E;

3. Algoritmo de Dijkstra

        graph TD;
            A[Inicio] --> B[Inicializar distancias con infinito];
            B --> C[Colocar distancia 0 en nodo de origen];
            C --> D[Agregar nodo origen a la cola de prioridad];
            D --> E[¿Cola vacía?];
            E -->|Sí| F[Fin];
            E -->|No| G[Extraer nodo con menor distancia];
            G --> H[Explorar vecinos];
            H --> I{¿Nueva distancia es menor?};
            I -->|Sí| J[Actualizar distancia y agregar a la cola];
            J --> H;
            I -->|No| E;

4. Algoritmo de Kruskal

        graph TD;
            A[Inicio] --> B[Ordenar aristas por peso];
            B --> C[Inicializar conjuntos disjuntos];
            C --> D[Iterar sobre las aristas ordenadas];
            D --> E{¿Mismo conjunto?};
            E -->|Sí| D;
            E -->|No| F[Unir conjuntos y agregar arista];
            F --> D;
            D --> G[Retornar árbol mínimo];
            G --> H[Fin];

5. Algoritmo de Floyd-Warshall

        graph TD;
            A[Inicio] --> B[Inicializar matriz de distancias];
            B --> C[Iterar sobre cada nodo intermedio];
            C --> D[Iterar sobre cada nodo de origen];
            D --> E[Iterar sobre cada nodo de destino];
            E --> F{¿Mejora la distancia con el intermedio?};
            F -->|Sí| G[Actualizar matriz de distancias];
            G --> E;
            F -->|No| C;
            C --> H[Retornar matriz final];
            H --> I[Fin];
